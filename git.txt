Памятка по Git
git clone “url” 		              > Клонируем
git add “filename” 		            > Добавляем файл новый
git add . 			                  > Добавляем (индексируем) все файлы и изменения
git commit -a -m "описание" 	    > делаем коммит (индексированных файлов)  с коментарием 
				                            ( -a-автоматическая индексация(все изменения), -m-коментарий )
git commit --amend 		            > Дописывает в последний commit
git status 			                  > изменение показывае	
git diff 			                    > сравнения 
git restore --staged <file>	      > отмена идексирования файла.  
git restore <file>
		            > отмена изменений в файле (берет с сервера).
git remote -v			                > просмотр удаленого репозиторя на сервере (-v - выводить путь)
remote add [сокращение] [url]	    > Добавляем репзиторий, `В дальнейшем можно обращаться к немучерез сокращение`

git fetch 'origin'		            > Забирает данные на локальный комп. Но не комитит или объединяет
git pull 			                    > Автоматически извлекает и затем сливает данные из удалённой ветки в вашу текущую ветку.

git ls-remote                     
git branch -a                     > Получение всех удаленных веток
  git checkout "ветвь"            > Переключаемся на ветку
  git fetch                       > Забираев ветку
  git pull 			                    > Автоматически извлекает и затем сливает данные из удалённой ветки в вашу текущую ветку.
